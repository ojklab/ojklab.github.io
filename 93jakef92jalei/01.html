let x1 = 50, x2 = 150, x3 = 250;
let side = 50;
let col = 255;
let y;

function setup() {
  createCanvas(350, 350);
  background(220);
  noLoop();
}

function draw() {
  y = 50;
  fill(col, 0, 0);
  square(x1, y, side);
  square(x2, y, side);
  square(x3, y, side);

  y = 100;
  fill(0, col, 0);
  square(x1, y, side);
  square(x2, y, side);
  square(x3, y, side);

  y = 150;
  fill(0, 0, col);
  square(x1, y, side);
  square(x2, y, side);
  square(x3, y, side);
}
