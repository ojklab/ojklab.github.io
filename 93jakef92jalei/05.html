function setup() {
  createCanvas(400, 100);
  background(220);
  noLoop();
}

function draw() {
  for (let i = 0; i < 8; i += 1) {
    if (i % 2 == 0) {
      fill(255, 127, 0); // 橙
    } else if (i % 3 == 0) {
      fill(0, 127, 255); // 青
    } else {
      fill(255);
    }
    circle(i * 50 + 25, height / 2, 50);
  }
}
