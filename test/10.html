function setup() {
  createCanvas(500, 100);
}

function draw() {
  background(220);

  const step = 50;

  for (let i = 1; i < 10; i += 1) {
    if (i % 2 == 0) {
      myCircle(i, step, false);
    } else {
      myCircle(i, step, true);
      myCircle(i, step / 2, false);
    }
  }
}

function myCircle(pos, dia, isBlue) {
  if (isBlue) {
    fill(0, 127, 255);  // 青
  } else {
    fill(255, 127, 0);  // 橙
  }
  circle(pos * 50, height / 2, dia);
}
