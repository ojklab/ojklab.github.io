let xpos = [];
 
function setup() {
  createCanvas(500, 150);
  noLoop();
  background(220);
  strokeWeight(2);
}
 
function draw() {
 
  for (let i = 1; i <= 4; i += 1) {
    let rand = floor(random(1, 5));  // 1～4の乱数を生成
    xpos.push(rand * 100);  // 描画する円のx座標を配列xposに追加
  }
 
  for (let idx = 0; idx < 4; idx += 1) {
    circle(/* 問11 */, height / 2, 100 - idx * 20);
  }
}
