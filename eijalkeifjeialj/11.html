function setup() {
  createCanvas(300, 100);
  background(220);
}

function draw() {
  for (let i = 0; i < 6; i += 1) {
    circle(calcX(i, 50), height / 2, 25);  // circle関数の第3引数を50→25
  }
}

function calcX(idx, dia) {
  let x = idx * dia + dia / 2;
  circle(x, height / 2, dia);  // 追加
  return x;
}
