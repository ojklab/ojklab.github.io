let R = 25;
let inDia = 15;

function setup() {
  createCanvas(500, 500);
  background(220);
}

function draw() {
  if (mouseIsPressed) {
    if (mouseButton == LEFT) {
      myCircle();
    } else {
      mySquare();
    }
  }
}

function myCircle() {
  noStroke();
  fill(0);
  circle(mouseX, mouseY, R * 2)
  orangeCircle();
}

function mySquare() {
  noStroke();
  fill(0);
  square(mouseX - R, mouseY - R, R * 2)
  blueCircle();
}

function orangeCircle() {
  fill(255, 127, 0);  // 橙
  circle(mouseX, mouseY, inDia);
}

function blueCircle() {
  fill(0, 127, 255);  // 青
  circle(mouseX, mouseY, inDia);
}
