let x;
let target;

function setup() {
  createCanvas(400, 100);
  x = width / 2;
  target = x;
}

function draw() {
  background(220);

  if (mouseIsPressed) {
    target = mouseX;
  }

  if (target - x > 5) {
    x += 5;
  } else if (target - x < -5) {
    x -= 5;
  } else {
    x = target;
  }

  circle(x, height / 2, 50);
}
