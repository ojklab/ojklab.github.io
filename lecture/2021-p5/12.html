/* クラス定義 */
class myCircle {
  constructor(_w, _h) {
    this.dia = 0;
    this.x = _w / 2;
    this.y = _h / 2;
    this.w = _w;
  }

  expansion() {
    this.dia += 5;
    if (this.dia > this.w) {
      this.dia = this.w;
    }
    circle(this.x, this.y, this.dia);
  }

  shrinkage() {
    this.dia -= 5;
    if (this.dia < 0) {
      this.dia = 0;
    }
    circle(this.x, this.y, this.dia);
  }
}


/* ここから本体 */
let mc;

function setup() {
  createCanvas(300, 300);
  background(220);
  mc = new myCircle(width, height);
}

function draw() {
  if (mouseIsPressed) {
    background(220);
    if (mouseButton == LEFT) {
      mc.expansion();
    } else if (mouseButton == RIGHT) {
      mc.shrinkage();
    }
    /* 問24 */
  }
}
